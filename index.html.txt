<script>
    const app = {
        products: [
            { id: 1, name: "Air Max Pulse", price: 15500, brand: "Nike", img: "https://images.unsplash.com/photo-1542291026-7eec264c27ff?w=800" },
            { id: 2, name: "UltraBoost 1.0", price: 18200, brand: "Adidas", img: "https://images.unsplash.com/photo-1587563871167-1ee9c731aefb?w=800" },
            { id: 3, name: "Air Jordan Retro", price: 24500, brand: "Nike", img: "https://images.unsplash.com/photo-1595950653106-6c9ebd614d3a?w=800" },
            { id: 4, name: "Adidas Yeezy 350", price: 28000, brand: "Adidas", img: "https://images.unsplash.com/photo-1584735175315-9d5df23860e6?w=800" },
            { id: 5, name: "Nike Pegasus 40", price: 16500, brand: "Nike", img: "https://images.unsplash.com/photo-1543508282-6319a3e4621f?w=800" },
            { id: 6, name: "Forum Low OG", price: 12000, brand: "Adidas", img: "https://images.unsplash.com/photo-1551107644-79bb0c58732e?w=800" }
        ],
        cart: [],
        wishlist: new Set(),

        init() {
            // Load from LocalStorage
            const savedCart = localStorage.getItem('kickz_cart');
            const savedWish = localStorage.getItem('kickz_wishlist');
            if(savedCart) this.cart = JSON.parse(savedCart);
            if(savedWish) this.wishlist = new Set(JSON.parse(savedWish));

            this.renderProducts(this.products);
            this.initCursor();
            this.updateUI();
            if(localStorage.getItem('theme') === 'dark') this.toggleTheme();
        },

        initCursor() {
            const cursor = document.getElementById('cursor');
            window.addEventListener('mousemove', (e) => {
                cursor.style.left = e.clientX + 'px';
                cursor.style.top = e.clientY + 'px';
            });
            document.addEventListener('mouseover', (e) => {
                if(e.target.closest('button, .card, .brand-logo, .filter-chip, .wishlist-btn, select')) {
                    cursor.style.transform = 'scale(3.5)';
                }
            });
            document.addEventListener('mouseout', (e) => {
                if(e.target.closest('button, .card, .brand-logo, .filter-chip, .wishlist-btn, select')) {
                    cursor.style.transform = 'scale(1)';
                }
            });
        },

        renderProducts(list) {
            const grid = document.getElementById('productGrid');
            grid.innerHTML = list.map((p, i) => `
                <div class="card" style="animation-delay: ${i*0.1}s">
                    <div class="wishlist-btn ${this.wishlist.has(p.id) ? 'active' : ''}" onclick="app.toggleWishlist(${p.id})">
                        ${this.wishlist.has(p.id) ? '❤️' : '🤍'}
                    </div>
                    <div class="img-container" onclick="app.openModal(${p.id})">
                        <img src="${p.img}" alt="${p.name}">
                    </div>
                    <h3>${p.name}</h3>
                    <p style="color:var(--primary); font-weight:bold;">Rs. ${p.price.toLocaleString()}</p>
                    <button class="btn-buy" onclick="app.addToCart(${p.id})">ADD TO CART</button>
                </div>
            `).join('');
        },

        sortProducts(criteria) {
            let sorted = [...this.products];
            if (criteria === 'price-low') sorted.sort((a, b) => a.price - b.price);
            if (criteria === 'price-high') sorted.sort((a, b) => b.price - a.price);
            this.renderProducts(sorted);
        },

        filterByBrand(brand, el) {
            document.querySelectorAll('.filter-chip').forEach(c => c.classList.remove('active'));
            el.classList.add('active');
            const filtered = brand === 'All' ? this.products : this.products.filter(p => p.brand === brand);
            this.renderProducts(filtered);
        },

        toggleWishlist(id) {
            if(this.wishlist.has(id)) this.wishlist.delete(id);
            else this.wishlist.add(id);
            
            localStorage.setItem('kickz_wishlist', JSON.stringify([...this.wishlist]));
            this.renderProducts(this.products); 
        },

        handleSearch(val) {
            const filtered = this.products.filter(p => 
                p.name.toLowerCase().includes(val.toLowerCase()) || 
                p.brand.toLowerCase().includes(val.toLowerCase())
            );
            this.renderProducts(filtered);
        },

        addToCart(id) {
            const existing = this.cart.find(item => item.id === id);
            if(existing) {
                existing.qty++;
            } else {
                const p = this.products.find(item => item.id === id);
                this.cart.push({...p, qty: 1});
            }
            this.saveCart();
            this.updateUI();
            this.showToast();
        },

        changeQty(id, delta) {
            const item = this.cart.find(i => i.id === id);
            if(item) {
                item.qty += delta;
                if(item.qty <= 0) this.cart = this.cart.filter(i => i.id !== id);
            }
            this.saveCart();
            this.updateUI();
        },

        saveCart() {
            localStorage.setItem('kickz_cart', JSON.stringify(this.cart));
        },

        updateUI() {
            const count = this.cart.reduce((sum, item) => sum + item.qty, 0);
            document.getElementById('cartCount').innerText = count;
            
            document.getElementById('cartList').innerHTML = this.cart.map(item => `
                <div style="display:flex; justify-content:space-between; align-items:center; padding:15px; background:var(--bg); border-radius:20px; margin-bottom:15px; box-shadow: 0 4px 10px var(--shadow);">
                    <div style="flex:1">
                        <b>${item.name}</b><br>
                        <small style="color:var(--primary)">Rs. ${(item.price * item.qty).toLocaleString()}</small>
                    </div>
                    <div class="qty-controls">
                        <span onclick="app.changeQty(${item.id}, -1)" style="cursor:pointer; padding:0 5px;">-</span>
                        <span>${item.qty}</span>
                        <span onclick="app.changeQty(${item.id}, 1)" style="cursor:pointer; padding:0 5px;">+</span>
                    </div>
                </div>
            `).join('');
            
            const total = this.cart.reduce((sum, i) => sum + (i.price * i.qty), 0);
            document.getElementById('cartTotal').innerText = total.toLocaleString();
        },

        toggleSearch() {
            const box = document.getElementById('searchBox');
            box.classList.toggle('active');
            if(box.classList.contains('active')) document.getElementById('searchInput').focus();
        },

        toggleTheme() {
            const body = document.body;
            const isDark = body.getAttribute('data-theme') === 'dark';
            body.setAttribute('data-theme', isDark ? 'light' : 'dark');
            localStorage.setItem('theme', isDark ? 'light' : 'dark');
        },

        toggleCart() { document.getElementById('cartSidebar').classList.toggle('open'); },
        
        openModal(id) {
            const p = this.products.find(item => item.id === id);
            document.querySelector('#modalImg img').src = p.img;
            document.getElementById('mTitle').innerText = p.name;
            document.getElementById('mBrand').innerText = p.brand;
            document.getElementById('mPrice').innerText = `Rs. ${p.price.toLocaleString()}`;
            document.getElementById('mBuyBtn').onclick = () => { this.addToCart(p.id); this.closeModal(); };
            document.getElementById('quickModal').classList.add('open');
        },

        closeModal() { document.getElementById('quickModal').classList.remove('open'); },
        
        showToast() {
            const t = document.getElementById('toast');
            t.classList.add('visible');
            setTimeout(() => t.classList.remove('visible'), 2000);
        },

        checkout() { 
            if(this.cart.length === 0) return alert("Your collection is empty!");
            alert("Order Placed! Get ready for your new KICKZ.");
            this.cart = [];
            this.saveCart();
            this.updateUI();
            this.toggleCart();
        }
    };

    app.init();
</script>
